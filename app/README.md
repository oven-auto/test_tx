<ul>

    <li>make docker-run</li>
    <li>make composer-install</li>
    <li>make make-env</li>
    <li>Настроить env-файл<br>
        <ul>
            <li>DB_CONNECTION=mysql
            <li>DB_HOST=localhost
            <li>DB_PORT=33062
            <li>DB_DATABASE=app
            <li>DB_USERNAME=app
            <li>DB_PASSWORD=secret
        </ul>
    </li>
    <li>make migrate</li>
</ul>
